    \section{输入/输出系统 \& 管理}

    \begin{enumerate}
        \item I/O系统层次结构：CPU $\rightarrow$ 通道 $\rightarrow$ 设备控制器 $\rightarrow$ 设备
        \begin{itemize}
            \item 在系统重，可能存在多个通道，每个通道可以连接多个控制器，每个控制器可以连接多个物理设备
            \item 通道负责控制多个设备控制器，减轻CPU的I/O负担
            \item 设备控制器负责控制设备并于CPU通信
            \item 设备负责执行I/O
        \end{itemize}
        \item 设备绝对号：系统为每个物理设备分配的唯一编号（物理设备号），用于标识具体的硬件设备
        \item I/O设备按信息交换的单位分类：
        \begin{itemize}
            \item 块设备：信息交换以数据块为单位。基本特征是传输速率较高、可寻址
            \item 字符设备：信息交换以字符为单位。基本特征是传输速率低、不可寻址，并且通常采用中断I/O方式
        \end{itemize}
        \item I/O设备按共享属性分类：
        \begin{itemize}
            \item 独占设备：同一时刻只能由一个进程占用的设备。通常采用静态分配方式
            \item 共享设备：{\color{red}{同一时间段}}内允许多个进程同时访问的设备。通常采用动态分配方式
            \begin{circlenum}
                \item 宏观上同时共享，微观上分时占用
                \item 通过分时的方式共享使用
                \item 必须是可寻址的和可随机访问的设备，从而保证数据的完整性和一致性，提高设备的利用率
            \end{circlenum}
            \item 虚拟设备：通过SPOOLing技术将独占设备改造为共享设备，将一个物理设备变为多个逻辑设备
        \end{itemize}
    \end{enumerate}

    \subsection{I/O接口}

    \begin{enumerate}
        \item I/O接口（I/O控制器）：主机和外设之间的交界界面，通过接口可以实现主机和外设之间的信息交换
        \item 磁盘驱动器由磁头、磁盘和读/写电路组成，故不属于I/O接口
    \end{enumerate}

    \subsubsection{I/O接口的基本结构}

    \begin{center}
        \includegraphics[width=0.8\linewidth]{\subfix{../../assets/408/operating-systems/5-1.png}}
    \end{center}

    \begin{enumerate}
        \item 设备控制器与CPU的接口：用于实现CPU与设备控制器之间的通信
        \begin{itemize}
            \item 数据线：传送读/写数据、状态信息、控制信息和中断类型号
            \item 地址线：传送要访问I/O接口中的寄存器的地址
            \item 控制线：传送读/写控制信号、中断请求和响应信号、握手信号
        \end{itemize}
        \item I/O逻辑：用于实现对设备的控制
        \item 设备控制器与设备的接口
        \item I/O端口：I/O接口电路中可被CPU直接访问的\textbf{寄存器}，主要有数据端口（读/写操作）、状态端口（读操作）和控制端口（写操作）
        \begin{itemize}
            \item 数据缓冲寄存器：暂存与CPU或内存之间传送的数据信息
            \item 状态寄存器：记录接口和设备的状态信息
            \item 控制寄存器：保存CPU对外设的控制信息
            \item 状态寄存器和控制寄存器在传送方向上是相反的，在访问时间上也是错开的，故合二为一
            \item I/O端口要想被CPU访问，就必须要对各个端口进行编址，每个端口对应一个端口地址，编址方式：
            \begin{circlenum}
                \item 独立编址（I/O映射方式）：对所有的I/O端口单独进行编址。I/O端口的地址空间与主存地址空间是两个独立的地址空间。需要设置专门的I/O指令来表明访问的是I/O地址空间
                \item 统一编制（存储器映射方式、内存映射I/O）：把主存地址空间分出一部分给I/O端口进行编址，根据地址范围就能区分访问是的I/O端口还是主存单元。使用指令系统中的{\color{red}{访存指令}}来完成输入/输出操作
            \end{circlenum}
        \end{itemize}
        \item I/O指令：对数据缓冲寄存器、状态/控制寄存器的访问操作是通过I/O指令完成的，I/O指令只能在操作系统内核的底层I/O软件中使用，是一种特权指令
    \end{enumerate}

    \subsubsection{I/O接口的功能}

    \begin{enumerate}
        \item 识别和接收命令
        \item 数据交换
        \item 标识和报告设备的状态
        \item 地址识别
        \item 数据缓冲：接口需设置数据缓冲寄存器，用于数据的暂存，以避免因速度不一致而丢失数据
        \item 差错控制
    \end{enumerate}

    \subsection{I/O控制方式}

    在输入/输出系统实现主机与I/O设备之间的{\color{red}{数据传送}}可以采用不同的控制方式：

    \begin{enumerate}
        \item 程序查询方式（程序直接控制方式、程序轮询方式）
        \item 程序中断方式（中断驱动方式）
        \item DMA方式（Direct Memory Access, 直接存储器存取）
        \item 通道方式：在系统中设有通道控制部件，每个通道都挂接若干外设，主机在执行I/O命令时，只需启动有关通道，通道将执行通道程序，从而完成I/O操作
    \end{enumerate}

    \subsubsection{程序中断（中断驱动方式）}

    \begin{enumerate}
        \item 程序中断：在计算机执行程序的过程中，出现某些继续处理的异常情况或特殊请求，CPU暂时中止现行程序，而转去对这些异常情况或特殊请求进行处理，处理完毕后再返回到原程序的断点处，继续执行原程序
        \item CPU与外设并行工作，传送与主程序串行工作
        \item CPU会在每个指令周期的末尾检查中断
        \item 数据传送单位是{\color{red}{字}}，每次读/写一个字
        \begin{itemize}
            \item Cache缺失：Cache完全是由硬件实现的，不会涉及中断层面
            \item 虚拟存储器时效：如缺页 $\Rightarrow$ 缺页中断（内中断）
            \item 浮点数运算下溢：当做机器零处理，不会引发中断
            \item 浮点数运算上溢：引发浮点溢出异常（内中断）
            \item CPU在执行一条指令的过程中检测异常事件
        \end{itemize}
        \item 若准备数据的时间小于中断响应和中断处理的时间，则数据被刷新，造成丢失
        \item 中断请求
        \begin{itemize}
            \item 中断源：请求CPU中断的设备或事件
            \item 中断请求标记触发器：记录中断时间并区分不同的中断源，其状态为$1$时，表示该中断源有请求
            \item 中断请求标记寄存器：由多个中断请求标记触发器组成
        \end{itemize}
        \item 中断{\color{red}{响应}}判优：CPU响应中断请求的先后顺序，通常通过{\color{customgreen}{硬件排队器}}或{\color{customgreen}{终端查询程序}}实现
        \begin{itemize}
            \item 中断优先级包括：
            \begin{circlenum}
                \item 响应优先级：由硬件线路或查询程序的查询顺序决定，不可动态改变
                \item 处理优先级：可利用中断屏蔽技术动态调整，以实现多重中断
            \end{circlenum}
            \item 可屏蔽中断：通过INRT线发出，优先级最低，在关中断模式下不被响应
            \item 不可屏蔽中断：通过NMI线发出，用于处理紧急和重要的事件，优先级最高
            \item 不可屏蔽中断 $>$ 可屏蔽中断
            \item 高速设备 $>$ 低速设备
            \item 输入设备 $>$ 输出设备
            \item 实时设备 $>$ 普通设备
        \end{itemize}
        \item CPU响应中断的条件
        \begin{itemize}
            \item 中断源有中断请求
            \item CPU允许中断及开中断（异常和不可屏蔽中断不受此限制）
            \item 一条指令执行完毕（异常不受此限制），且没有更紧迫的任务
        \end{itemize}
        \item 中断隐指令：CPU响应中断后，经过某些操作，转去执行中断服务程序，这些操作是有硬件直接实现的，即中断隐指令，其本质是{\color{customgreen}{CPU内部硬件控制逻辑自动完成的一系列操作}}，所完成的操作：
        \begin{itemize}
            \item 关中断：CPU响应中断后，首先要保护程序的断点和现场信息，在保护断点和现场的过程中，CPU不能响应更高级中断源的中断请求
            \item 保存断点：将原程序的断点保存在栈或特定寄存器中
            \item 引出中断程序：识别中断源，将对应的服务程序入口地址送入程序计数器PC。识别中断源方法
            \begin{circlenum}
                \item 硬件向量法/向量中断
                \item 软件查询法/非向量中断
            \end{circlenum}
            \item 中断向量：每个中断源都有一个唯一的类型号，每个中断类型号都对应一个中断服务程序，每个中断服务程序都有一个入口地址，即中断向量
            \item 中断向量地址：中断向量表的地址，也是中断服务程序入口地址的地址
            \item 中断向量表：系统中的全部中断向量
            \item 中断向量法：CPU响应中断后，通过识别中断源获得中断类型号，然后据此计算出对应中断向量的地址，再根据地址从中断向量表中取出中断服务程序的入口地址，并送入程序计数器PC，以转去执行中断服务程序
            \item 向量中断：采用中断向量法的中断
        \end{itemize}
        \item 中断处理过程 \\
        \begin{minipage}[t]{0.64\textwidth}
            \vspace{0pt}
            \begin{itemize}
                \item 前三步由中断隐指令（硬件自动）完成
                \item 第四步之后都是由中断服务程序完成
                \item 第五步开中断：允许更高级中断请求得到响应，已实现中断嵌套
                \item 最后一步：中断返回指令，需要修改PC值，并且要将CPU中的所有寄存器都恢复到中断前的状态
            \end{itemize}
        \end{minipage}
        \hfill
        \begin{minipage}[t]{0.32\textwidth}
            \vspace{0pt}
            \includegraphics[width=\linewidth]{\subfix{../../assets/408/computer-organization-and-architecture/7-4.png}}
        \end{minipage}
        \item 单重中断：在CPU执行中断服务程序的过程中，若又出现了新的优先级更高的中断请求，而CPU对新的中断请求不予响应
        \item 多重中断（中断嵌套）：若CPU暂停现行的中断服务程序，转去处理新的中断请求
        \item {\color{red}{中断处理优先级}}：多重中断的实际优先级处理次序，可以利用中断屏蔽技术动态调整。通过在中断系统中设置{\color{red}{中断屏蔽字}}寄存器实现。每个中断源对应中断屏蔽字寄存器中的某一位
        \begin{itemize}
            \item 1：表示屏蔽该中断源的请求
            \item 0：表示可以正常请求
            \item 中断屏蔽字：中断屏蔽字寄存器的内容
            \item 中断屏蔽字越多1，优先级越高
            \item 每个屏蔽字至少有一个1 $\Rightarrow$ 屏蔽自身中断
            \item 所有中断请求信号都会与中断屏蔽字寄存器中对应位的非值进行逻辑与，然后被送入中断判优电路
            \item 通过配置中断屏蔽字，可以动态改变中断处理优先级
        \end{itemize}
    \end{enumerate}

    \subsubsection{DMA方式（Direct Memory Access, 直接存储器存取）}

    \begin{enumerate}
        \item DMA方式（直接存储器存取方式）：完全由硬件进行成组信息传送的控制方式，具有程序中断方式的优点（即在数据准备阶段，CPU与外设并行工作），外设和内存之间开辟了一条直接数据通路，传送信息不再经过CPU
        \item 特点
        \begin{itemize}
            \item 基本传送单位是数据块，而不再是字
            \item 所传送的数据，是从设备直接送入内存的，或者相反，而不再经过CPU
            \item 仅在传送一个或多个数据块的开始和结束时，才需要CPU干预
        \end{itemize}
        \item 只有具有DMA接口的设备才能产生DMA请求
        \item {\color{red}{DMA优先级高于中断请求}}
        \item $\color{red}{\bigstar\bigstar}$CPU会在每个机器周期（总线事务、存取周期、总线周期、当前流水段长度）结束后检查是否有DMA请求以及响应
        \item I/O与主机并行工作，程序与传送并行工作
        \item DMA控制器（DMA接口）：对数据传送过程中进行控制的硬件
        \begin{itemize}
            \item 当I/O设备需要进行数据传送时，{\color{red}{（外部设备）向DMA控制器发出DMA请求信号}}，通过DMA控制器向CPU提出DMA传送请求，CPU响应之后将让出系统总线，有DMA控制器接管总线进行数据传送 $\Rightarrow$ DMA控制器具有控制相同总线的能力
            \item 组成
            \includegraphics[width=\linewidth]{\subfix{../../assets/408/computer-organization-and-architecture/7-8.png}}
            \begin{circlenum}
                \item CR（Command Register，命令/状态寄存器）：用于存放CPU发来的I/O命令或设备的状态信息
                \item MAR（Memory Address Register, 主存地址计数器、内存地址寄存器）：存放交换数据的主存地址
                \item DR（Data Register，数据寄存器、数据缓冲寄存器）：暂存每次传送的数据。{\color{customgreen}{DMA接口与主存之间传送单位为字}}
                \item DC（Data Counter，传送长度计数器、数据计数器）：记录传送数据的长度
                \item DMA请求触发器
                \item 控制/状态逻辑：用于指定传送方向，修改传送参数，并对DMA请求信号、CPU响应信号进行协调和同步
                \item 中断机构：当一批数据传送完毕后触发中断机构，向CPU提出中断请求
            \end{circlenum}
        \end{itemize}
        \item DMA的传送方式：当I/O设备和CPU同时访问主存时，可能发生冲突，避免冲突方法
        \begin{itemize}
            \item 停止CPU访存
            \item 周期挪用：由于I/O访存的有优先级高于CPU访存（I/O不立即访存则可能丢失数据），I/O设备挪用一个存取周期，传送完一个{\color{red}{数据字}}后立即释放总线。{\color{red}{单字传送方式}}。I/O设备有DMA请求时，有以下情况
            \begin{circlenum}
                \item CPU不在访存
                \item CPU正在访存：等待存取周期结束，CPU再将总线占有权让出
                \item 同时请求访存：CPU要暂时放弃总线占有权
            \end{circlenum}
            \item 交替访存：将CPU的工作周期分成两个时间片，一个给CPU访存，另一个给DMA访存。适用于CPU的工作周期比主存存取周期长的情况
        \end{itemize}
        \item DMA传送过程
        \begin{itemize}
            \item 预处理：由CPU完成一些必要的准备工作。运行在内核态。由请求I/O的进程负责，故请求I/O的进程处于运行态
            \begin{circelnum}
                \item 初始化DMA控制器中的有关寄存器、设置传送方向、测试并启动设备等
                \item CPU继续执行原程序，直到I/O设备准备好发送的数据（输入情况）或接受的数据（输出情况）时，I/O设备向DMA控制器发送DMA请求，再由DMA控制器向CPU发出总线请求
            \end{circelnum}
            \item 数据传送：DMA（硬件）完成，{\color{red}{DMA以数据块为基本传送单位}}（CPU 一次交给 DMA 的任务是多少）
            \item 后处理：DMA控制器向CPU发出中断请求，CPU执行中断服务程序做DMA结束处理。运行在内核态。此时请求I/O的进程处于阻塞态
        \end{itemize}
    \end{enumerate}

    \subsubsection{计算}

    \begin{enumerate}
        \item 由于端口有限，故必须在外设传输完端口大小的数据时访问端口，以防止数据未被及时读取而丢失
        \item 外设准备数据时间 = $\dfrac{\text{外设准备的数据量}}{\text{数据率}}$
        \item CPU用于外设I/O的时间占CPU总时间的百分比 = $\dfrac{\text{每秒CPU用于外设I/O的时钟周期}}{\text{主频}}$
        \item 外设每秒DMA次数 = $\dfrac{\text{数据率}}{\text{每次DMA传送块大小}}$
    \end{enumerate}

    \end{center}

    \begin{enumerate}
        \item 用户层软件
        \item 设备独立性软件（设备无关性）：实现用户程序与设备驱动器的统一接口、设备命名、设备保护，以及设备的分配与释放等，同时为设备管理和数据传送提供必要的存储空间
        \begin{itemize}
            \item 让用户程序不用关心具体设备，程序只面对逻辑设备，设备独立性软件负责逻辑设备与物理设备的映射
            \item 主要功能：
            \begin{circlenum}
                \item 执行所有设备的公有操作
                \item 向用户层（或文件层）提供统一接口
            \end{circlenum}
        \end{itemize}
        \item 设备驱动程序：与硬件直接相关，{\color{customgreen}{每类设备需要配置一个设备驱动程序}}
        \begin{itemize}
            \item 负责具体实现系统对设备发出的操作指令，驱动I/O设备工作的驱动程序，是I/O进程和设备控制器之间的通信程序
            \item 通常以{\color{red}{进程}}的形式存在
        \end{itemize}
        \item 中断处理程序：和硬件交互
    \end{enumerate}

    \subsubsection{逻辑设备名 VS 物理设备名}

    \begin{enumerate}
        \item 应用程序所用的设备不局限于某个具体的物理设备
        \item 在应用程序中，使用逻辑设备名来请求使用某类设备，而在系统实际执行时，必须将逻辑设备名映射成物理设备名
        \item 逻辑设备名好处：
        \begin{itemize}
            \item 增加设备分配的灵活性
            \item 易于实现I/O重定向（I/O重定向：用于I/O操作的设备可以更换，而不必改变应用程序）
        \end{itemize}
        \item 逻辑设备表（Logical Unit Table, LUT）：逻辑设备名与物理设备名的映射
        \begin{itemize}
            \item LUT = 逻辑设备名 + 物理设备名 + 设备驱动程序的入口地址
            \item 两种方式设置 LUT：
            \begin{circlenum}
                \item 整个系统重只设置一张 LUT
                \item 为每个用户设置一张 LUT
            \end{circlenum}
        \end{itemize}
    \end{enumerate}

    \subsubsection{设备分配}

    \begin{enumerate}
        \item 数据结构
        \begin{itemize}
            \item 设备控制表（DCT）：每个设备对应一张DCT，表中的表项就是设备的各个属性
            \item 控制器控制表（COCT）：每个设备控制器都对应一张COCT，操作系统根据COCT的信息对控制器进行操作和管理
            \item 通道控制表（CHCT）：每个通道对应一张CHCT，操作系统根据CHCT的信息对通道进行操作和管理
            \item 系统设备表（SDT）：整个系统只有一张SDT，记录已连接到系统重的所有物理设备的情况，每个物理设备对应一个表目
        \end{itemize}
        \item 设备分配时应考虑的因素
        \begin{itemize}
            \item 设备的固有属性：独占设备 Or 共享设备 Or 虚拟设备
            \item 设备的分配算法：FCFS算法 Or 最高优先级优先算法
            \item 设备分配中的安全性
            \begin{circlenum}
                \item 安全分配方式：每当进程发出I/O请求后，便进入阻塞态，知道其I/O操作完成时才被唤醒
                \item 不安全分配方式：进程在发出I/O请求后仍继续运行，需要是会继续发出I/O请求，仅当进程所请求的设备已被另一进程占用时，才进入阻塞态
            \end{circlenum}
        \end{itemize}
        \item 步骤
        \begin{itemize}
            \item 分配设备
            \item 分配控制器
            \item 分配通道
            \item 只有设备、控制器和通道都分配成功，设备分配才算成功，之后便可启动设备进行数据传送
        \end{itemize}
    \end{enumerate}

    \subsection{应用程序I/O接口}

    \begin{enumerate}
        \item 应用程序I/O接口是软件和操作系统之间的接口，I/O接口（I/O控制器）是CPU/内存和外部设备之间的硬件接口
        \item 分类：
        \begin{itemize}
            \item 字符设备接口
            \begin{circelnum}
                \item 字符设备：数据的存取和传输是以字符为单位的设备
                \item 基本特征：传输速率较低、不可寻址，并且在输入/输出是通常采用{\color{red}{中断驱动方式}}
            \end{circelnum}
            \item 块设备接口
            \begin{circlenum}
                \item 块设备：数据的存取和传输是以数据块为单位的设备
                \item 基本特征：传输速率较高，可寻址。磁盘设备的I/O通常采用{\color{red}{DMA方式}}
            \end{circlenum}
            \item 网络设备接口：网络I/O接口为网络套接字接口
        \end{itemize}
        \item 模式
        \begin{itemize}
            \item 阻塞I/O：当用户进程调用I/O操作时，进程就被阻塞，并移到阻塞队列，I/O操作完成后，进程才被唤醒，移到就绪队列
            \item 非阻塞I/O：当用户进程调用I/O操作是，不阻塞该进程，但进程需要不断询问I/O操作是否完成，在I/O执行阶段，进程还可以做其他事情
        \end{itemize}
    \end{enumerate}

    \subsection{缓冲管理}

    \begin{enumerate}
        \item 磁盘高速缓存技术：利用内存中的存储空间来暂存从磁盘中读出的一系列盘块中的信息。磁盘高速缓存逻辑上属于磁盘，物理上是驻留在内存中的盘块
    \end{enumerate}

    \subsubsection{缓冲区（Buffer）}

    \begin{enumerate}
        \item 缓冲区是一种临界资源，在使用缓冲区时是互斥访问的
        \item 引入缓冲区的目的
        \begin{itemize}
            \item 缓和CPU与I/O设备间速度不匹配的矛盾
            \item 减少对CPU的中断频率，放宽对CPU中断响应时间的限制
            \item 解决基本数据单元大小不匹配的问题
            \item 提高CPU和I/O设备之间的并行性
        \end{itemize}
        \item 缓冲区的实现方法
        \begin{itemize}
            \item 采用硬件缓冲器
            \item 利用内存作为缓冲区
        \end{itemize}
        \item 根据系统设置缓冲区的个数，缓冲技术分类：
        \begin{itemize}
            \item T：从设备将一块数据输入缓冲区的时间
            \item M：操作系统将该缓冲区中的数据传送到工作区的时间
            \item C：CPU对这一块数据进行处理的时间
            \item 单缓冲：每当用户进程发出一个I/O请求，操作系统便在内存中为之分配一个缓冲区
            \begin{circlenum}
                \item 一个缓冲区的大小就是一个块
                \item 必须等缓冲区装满后才能从缓冲区取走数据
                \item T 和 C 是可以并行的
                \item 单缓冲区处理每块数据的平均时间：$Max(C, T) + M$
                \item 缓冲区是共享资源，故使用时必须互斥
            \end{circlenum}
            \item 双缓冲（缓冲对换）
            \begin{circlenum}
                \item C 和 M 是可以与 T 并行的
                \item 双缓冲区处理每块数据的平均时间：$Max(C + M, T)$
                \item 可以实现双向数据传输
                \item 若输入和输出的速度相差较大时，效果不理想
            \end{circlenum}
            \item 循环缓冲（多缓冲机制）：包含多个大小相等的缓冲区，多个缓冲区链接称一个循环队列
            \item 缓冲池：包含一个用于管理自身的数据结构和一组操作函数的管理机制，用于管理多个缓冲区
            \begin{circlenum}
                \item 缓冲区按其使用状况分为
                \begin{enumerate}
                    \item 空缓冲队列
                    \item 输入队列
                    \item 输出队列
                \end{enumerate}
                \item 4中工作缓冲区
                \begin{enumerate}
                    \item hin：用于收容输入数据的工作缓冲区
                    \item sin：用于提取输入数据的工作缓冲区
                    \item hout：用于收容输出数据的工作缓冲区
                    \item sout：用于提取输出数据的工作缓冲区
                \end{enumerate}
                \item 4种工作方式
                \begin{enumerate}
                    \item 收容输入
                    \item 提取输入
                    \item 收容输出
                    \item 提取输出
                \end{enumerate}
            \end{circlenum}
        \end{itemize}
    \end{enumerate}

    \subsubsection{SPOOLing技术（假脱机技术）}

    \begin{enumerate}
        \item 目的：缓和CPU的高速型与I/O设备的低速性之间的矛盾，将{\color{red}{独占设备改造成共享设备}}的技术
        \item 多道程序技术（Multiprogramming）：让CPU和I/O设备尽量同时工作，提高系统资源利用率
        \item 多道程序技术是 SPOOLing 技术的基础。SPOOLing 利用多道程序技术，使输入进程、输出进程和用户进程并发执行，并借助磁盘作为缓冲区，将独占设备虚拟为共享设备，从而提高 CPU 和设备的利用率
        \item SPOOLing技术是一种{\color{customgreen}{以空间换时间}}的技术
        \item SPOOLing技术：利用专门的外围控制机，先将低速I/O设备上的数据传送到高速磁盘上，或者相反，当CPU需要输入数据时，便可直接从磁盘中读取数据；反之，当CPU需要输出数据时，也能以很快的速度将数据先输出到磁盘上
        \item 例子：扫描仪、打印机
        \item 组成
        \begin{itemize}
            \item 输入井和输出井
            \item 输入缓冲区和输出缓冲区
            \item 输入进程和输出进程
        \end{itemize}
        \item 核心程序
        \begin{itemize}
            \item 预输入程序
            \item 缓输出程序
            \item 井管理程序
        \end{itemize}
    \end{enumerate}
