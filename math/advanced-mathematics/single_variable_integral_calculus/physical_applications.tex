    \section{一元函数积分学的应用 - 物理应用}

    \subsection{变力沿直线做功}

    设方向沿$x$轴正向的力函数为$y = F(x)(a \leq x \leq b)$，则物体沿$x$轴从点$a$移动到点$b$时，变力$F(x)$所做的功为
    \[
        W = \int_a^b F(x)\,\mathrm{d}x
    \]
    功的微元$dW = F(x)\,\mathrm{d}x$

    \subsection{抽水做功}

    \begin{center}
        \begin{minipage}{0.58\textwidth}
            如图所示，将容器中的水全部抽出所做的功为
            \[
                W = \rho g\int_a^b xA(x)\,\mathrm{d}x
            \]
            其中$\rho$为水的密度，$g$为重力加速度。 \\

            功的微元$dW = \rho gA(x)\,\mathrm{d}x \cdot x = \rho gA(x)\cdot x\,\mathrm{d}x$为位于$x$处厚度为$\mathrm{d}x$，水平截面面积为$A(x)$的一层水被抽出（路程为$x$）所做的功

            \medskip
            \textbf{物理公式}
            \begin{enumerate}
                \item 功：$W = F\cdot s = mg\cdot s = \rho vg\cdot s = \rho Shg\cdot s$
            \end{enumerate}

    \subsection{静水压力（水压力）}

    \begin{center}
        \begin{minipage}{0.58\textwidth}
            如图所示，垂直浸没在水中的平板$ABCD$的一侧受到的水压力为
            \[
                P = \rho g\int_a^b x[f(x) - g(x)]\,\mathrm{d}x
            \]
            其中$\rho$为水的密度，$g$为重力加速度。\\

            压力微元$dP = \rho gx[f(x) - g(x)]\,\mathrm{d}x$，即图中矩形条所收到的压力，$x$表示水深，$[f(x) - g(x)]$表示矩形条的宽度，$\mathrm{d}x$是矩形条的高度

            \medskip
            \textbf{物理公式}
            \begin{enumerate}
                \item 静水压强公式: $p = \rho gh$
                \item 压力公式: $F = p \cdot S$
            \end{enumerate}

    \subsection{引力}

    \begin{center}
        \begin{minipage}{0.58\textwidth}
            如图所示，设有一长度为$l$、线密度为常数$\mu$的细棒，在与细棒右端的距离为$a$处有以质量为$m$的质点$M$，已知引力常量为$G$，则质点$M$与细棒之间的引力的大小为
            \[
                F = \int_{-l}^0 \frac{Gm\mu}{(a - x)^2}\,\mathrm{d}x
            \]
            \medskip
            \textbf{物理公式}
            \begin{enumerate}
                \item 万有引力定律：$F = \displaystyle\frac{G m_1 m_2}{r^2}$
                \item 线密度定义：
                \[
                    \mu = \frac{\mathrm{d}m}{\mathrm{d}x}
                    \quad \Longrightarrow \quad
                    \mathrm{d}m = \mu\,\mathrm{d}x
                \]
            \end{enumerate}

    \subsection{古鲁金第一定理：旋转体的侧面积}

    设平面光滑曲线弧$L$的长度为$l$，其形心坐标为$C(\bar{x}, \bar{y})$。若曲线弧$L$绕其所在平面内一条{\color{red}{不与曲线弧相交}}的定直线旋转一周，则所得旋转曲面的侧面积为
    \[
        S = 2\pi l \cdot r(\bar{x}, \bar{y})
    \]
    其中$r(\bar{x}, \bar{y})$为形心到旋转轴的距离

    \medskip
    \begin{enumerate}
        \item 若曲线弧$L$绕$x$轴旋转一周，则$S = 2\pi l |\bar{y}|$
        \item 若曲线弧$L$绕$y$轴旋转一周，则$S = 2\pi l |\bar{x}|$
        \item 一般地，若绕某条定直线旋转，则关键是先求{\color{red}{形心到该直线的距离}}$d$
    \end{enumerate}

    \subsection{古鲁金第二定理：旋转体的体积}

    设平面图形$D$的面积为$A$，其形心坐标为$C(\bar{x}, \bar{y})$。若平面图形$D$绕其所在平面内一条{\color{red}{不与该图形相交}}的定直线旋转一周，则所得{\color{red}{旋转体的体积}}为
    \[
        V = 2\pi r(\bar{x}, \bar{y})\cdot A
    \]
    其中$r(\bar{x}, \bar{y})$为平面图形形心到旋转轴的距离

    \medskip
    \begin{enumerate}
        \item 若平面图形$D$绕$x$轴旋转一周，则$V = 2\pi |\bar{y}|\,A$
        \item 若平面图形$D$绕$y$轴旋转一周，则$V = 2\pi |\bar{x}|\,A$
        \item 一般地，若绕某条定直线旋转，则关键仍是先求{\color{red}{形心到旋转轴的距离}}$d$
    \end{enumerate}

    \subsection{结论}

    \begin{enumerate}
        \item 圆的问题将圆心放在坐标原点，此时圆方程最简
    \end{enumerate}

    \subsection{理解}

    \begin{enumerate}
        \item 古鲁金第二定理 VS. 10.2 用定积分表达和计算旋转体的体积
        \begin{analysisbox}
            设平面区域
            \[
                D = \{(x, y) \mid 0 \le y \le f(x),\ a \le x \le b\}
            \]
            则
            \[
                A = \int_a^b f(x)\,\mathrm{d}x
            \]
            形心坐标（10.4 平面上的曲边梯形的形心坐标公式）为：
            \[
                \bar{x} = \frac{\displaystyle\int_a^b x f(x)\,\mathrm{d}x}{\displaystyle\int_a^b f(x)\,\mathrm{d}x},
                \qquad
                \bar{y} = \frac{\displaystyle\frac12\int_a^b [f(x)]^2\,\mathrm{d}x}{\displaystyle\int_a^b f(x)\,\mathrm{d}x}
            \]

            所以
            \begin{enumerate}
                \item 绕$x$轴旋转一周
                \[
                    V = 2\pi \bar{y}A = 2\pi \cdot \frac{\frac12\dint_a^b [f(x)]^2\,\mathrm{d}x}{\dint_a^b f(x)\,\mathrm{d}x}\cdot \int_a^b f(x)\,\mathrm{d}x = \pi\int_a^b [f(x)]^2\,\mathrm{d}x
                \]
                \item 绕$y$轴旋转一周
                \[
                    V = 2\pi \bar{x}A = 2\pi \int_a^b x f(x)\,\mathrm{d}x
                \]
            \end{enumerate}

        \end{analysisbox}
    \end{enumerate}
