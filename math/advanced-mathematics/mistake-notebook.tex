    \section{错题}

    \subsection{一元函数微分学的概念}

    \subsubsection{题目}
    \begin{enumerate}
        \item 设$F(x) = g(x)\varphi(x)$，$\varphi(x)$在$x = a$处连续但不可导，$g(x)$在$x = a$处可导，$F(x)$在$x = a$处可导，则一定有
        \item 设可导函数$f(x) > 0$，求$\lim\limits_{n\to \infty} n\ln\dfrac{f(\dfrac{1}{n})}{f(0)}$
        \item 设$f(x) = \begin{cases}
                            x^3 \sin \dfrac{1}{x}, x > 0\\
                            x^2, x \leq 0
        \end{cases}$，则$f(x)$在$x = 0$处
        \item 设$f(x) = \lim\limits_{n\to \infty}\dfrac{x^2 e^{n(x-1)} + ax + b}{5 + e^{n(x-1)}}$，求$f(x)$并讨论$f(x)$的连续性及可导性与$a, b$的关系
    \end{enumerate}

    \subsubsection{解答}

    \subsubsection*{第1题解答}
    \begin{analysisbox}
        \textbf{结论：}$g(a) = 0$，且 $F'(a) = g'(a)\varphi(a)$。

        \medskip

        \textbf{证明：}由导数定义，
        \[
            F'(a) = \lim_{x\to a} \frac{F(x) - F(a)}{x - a}
            = \lim_{x\to a} \frac{g(x)\varphi(x) - g(a)\varphi(a)}{x - a}
        \]
        添加并减去 $g(a)\varphi(x)$：
        \[
            \begin{aligned}
                F'(a)
                &= \lim_{x\to a} \left[ \varphi(x) \cdot \frac{g(x) - g(a)}{x - a} + g(a) \cdot \frac{\varphi(x) - \varphi(a)}{x - a} \right] \\[0.5em]
                &= \varphi(a) \cdot g'(a) + g(a) \cdot \underbrace{\lim_{x\to a} \frac{\varphi(x) - \varphi(a)}{x - a}}_{\text{不存在}}
            \end{aligned}
        \]
        由于 $\varphi(x)$ 在 $x=a$ 处不可导，上式中画线部分极限不存在。而已知 $F'(a)$ 存在，因此必须 $g(a) = 0$。

        \medskip

        此时 $F'(a) = g'(a)\varphi(a)$。
    \end{analysisbox}

    \subsubsection*{第2题解答}
    \begin{analysisbox}[解]
        令 $x = \dfrac{1}{n}$，当 $n \to \infty$ 时 $x \to 0^+$。
        \[
            \begin{aligned}
                \lim_{n\to\infty} n\ln\frac{f\!\left(\dfrac{1}{n}\right)}{f(0)}
                &= \lim_{x\to 0^+} \frac{1}{x} \ln\frac{f(x)}{f(0)} \\[0.5em]
                &= \lim_{x\to 0^+} \frac{\ln f(x) - \ln f(0)}{x - 0} \\[0.5em]
                &= \left.[\ln f(x)]'\right|_{x=0} = \frac{f'(0)}{f(0)}
            \end{aligned}
        \]

            {\color{red}{注}}：$\ln \dfrac{b}{a} = \ln b - \ln a$
    \end{analysisbox}

    \subsubsection*{第3题解答}
    \begin{analysisbox}
        \textbf{（1）连续性}
        \[
            f(0) = 0^2 = 0, \quad
            \lim_{x\to 0^+} f(x) = \lim_{x\to 0^+} x^3\sin\dfrac{1}{x} = 0, \quad
            \lim_{x\to 0^-} f(x) = \lim_{x\to 0^-} x^2 = 0,
        \]
        故 $\lim\limits_{x\to 0} f(x) = 0 = f(0)$，$f(x)$ 在 $x=0$ 处{\color{red}{连续}}。

        \textbf{（2）可导性}

        \[
            f'_+(0) = \lim_{x\to 0^+} \dfrac{x^3\sin\dfrac{1}{x} - 0}{x - 0}
            = \lim_{x\to 0^+} x^2\sin\dfrac{1}{x} = 0,
        \]
        \[
            f'_-(0) = \lim_{x\to 0^-} \dfrac{x^2 - 0}{x - 0}
            = \lim_{x\to 0^-} x = 0,
        \]
        $f'_+(0) = f'_-(0) = 0$，故 $f(x)$ 在 $x=0$ 处{\color{red}{可导}}，且 $f'(0) = 0$。

        \textbf{（3）导函数连续性}

        当 $x > 0$ 时，$f'(x) = 3x^2\sin\dfrac{1}{x} - x\cos\dfrac{1}{x}$；当 $x < 0$ 时，$f'(x) = 2x$。
        \[
            \lim_{x\to 0^+} f'(x) = \lim_{x\to 0^+}\left(3x^2\sin\dfrac{1}{x} - x\cos\dfrac{1}{x}\right) = 0,
        \]
        \[
            \lim_{x\to 0^-} f'(x) = \lim_{x\to 0^-} 2x = 0,
        \]
        故 $\lim\limits_{x\to 0} f'(x) = 0 = f'(0)$，$f'(x)$ 在 $x=0$ 处{\color{red}{连续}}。

        \textbf{（4）二阶可导性}

        当 $x > 0$ 时，
        \[
            f'(x) = 3x^2\sin\dfrac{1}{x} - x\cos\dfrac{1}{x},
        \]
        \[
            \begin{aligned}
                f''_+(0) &= \lim_{x\to 0^+} \dfrac{f'(x) - f'(0)}{x - 0} \\[0.5em]
                &= \lim_{x\to 0^+} \dfrac{3x^2\sin\dfrac{1}{x} - x\cos\dfrac{1}{x}}{x} \\[0.5em]
                &= \lim_{x\to 0^+}\left(3x\sin\dfrac{1}{x} - \cos\dfrac{1}{x}\right),
            \end{aligned}
        \]
        其中 $3x\sin\dfrac{1}{x} \to 0$，但 $\cos\dfrac{1}{x}$ 振荡无极限，故 $f''_+(0)$ {\color{red}{不存在}}。

        \medskip

        \textbf{综上，}$f(x)$ 在 $x=0$ 处连续、可导，且导函数连续，但{\color{red}{不可二次可导}}。
    \end{analysisbox}

    \subsubsection*{第4题解答}
    \begin{analysisbox}[解]
        \textbf{第一步：求 $f(x)$}

        关键在于 $e^{n(x-1)}$ 当 $n \to \infty$ 时的行为：
        \begin{itemize}
            \item $x > 1$ 时，$e^{n(x-1)} \to +\infty$；
            \item $x = 1$ 时，$e^{n(x-1)} = 1$；
            \item $x < 1$ 时，$e^{n(x-1)} \to 0$。
        \end{itemize}

        \medskip

        \textbf{当 $x > 1$}：分子分母同除以 $e^{n(x-1)}$，
        \[
            f(x) = \lim_{n\to\infty} \frac{x^2 + \dfrac{ax+b}{e^{n(x-1)}}}{\dfrac{5}{e^{n(x-1)}} + 1} = \frac{x^2 + 0}{0 + 1} = x^2
        \]

        \textbf{当 $x = 1$}：
        \[
            f(1) = \frac{1 + a + b}{5 + 1} = \frac{1 + a + b}{6}
        \]

        \textbf{当 $x < 1$}：
        \[
            f(x) = \lim_{n\to\infty} \frac{0 + ax + b}{5 + 0} = \frac{ax + b}{5}
        \]

        综上，
        \[
            f(x) = \begin{cases}
                       x^2, & x > 1 \\[0.3em]
                       \dfrac{1+a+b}{6}, & x = 1 \\[0.3em]
                       \dfrac{ax+b}{5}, & x < 1
            \end{cases}
        \]

        \medskip

        \textbf{第二步：连续性}

        $x \neq 1$ 时 $f(x)$ 显然连续，只需考察 $x = 1$ 处：
        \[
            \lim_{x\to 1^-} f(x) = \frac{a+b}{5}, \quad
            \lim_{x\to 1^+} f(x) = 1, \quad
            f(1) = \frac{1+a+b}{6}
        \]

        连续的充要条件是三者相等，即
        \[
            \frac{a+b}{5} = 1 \quad\Longrightarrow\quad a + b = 5
        \]

        验证：$a+b=5$ 时 $f(1) = \dfrac{1+5}{6} = 1$，三者一致。

        \medskip

        {\color{red}{结论：}}当 $a + b = 5$ 时 $f(x)$ 在 $\mathbb{R}$ 上连续；当 $a + b \neq 5$ 时 $f(x)$ 在 $x=1$ 处不连续。

        \medskip

        \textbf{第三步：可导性}

        $x \neq 1$ 时 $f(x)$ 显然可导。当 $a+b=5$（连续条件下），考察 $x=1$ 处的可导性：

        \[
            f'_-(1) = \lim_{x\to 1^-} \frac{ax+b}{5}\bigg|' = \frac{a}{5}, \quad
            f'_+(1) = \lim_{x\to 1^+} (x^2)' = 2
        \]

        可导的充要条件是 $f'_-(1) = f'_+(1)$，即
        \[
            \frac{a}{5} = 2 \quad\Longrightarrow\quad a = 10, \quad b = 5 - 10 = -5
        \]

        \medskip

        {\color{red}{综上：}}
        \begin{itemize}
            \item 当 $a + b \neq 5$ 时，$f(x)$ 在 $x=1$ 处{\color{red}{不连续}}，自然不可导；
            \item 当 $a + b = 5$ 但 $a \neq 10$ 时，$f(x)$ 在 $\mathbb{R}$ 上{\color{red}{连续但不可导}}；
            \item 当 $a = 10$，$b = -5$ 时，$f(x)$ 在 $\mathbb{R}$ 上{\color{red}{连续且可导}}。
        \end{itemize}
    \end{analysisbox}

    \subsubsection{解答}

    \subsubsection*{第1题解答}
    \begin{analysisbox}
        将绝对值写成根式：
        \[
            y(x)=|f(x-1)|=\sqrt{[f(x-1)]^2}.
        \]
        因$f$可导，故$f$连续；又$f(0)=-1\neq 0$，所以当$x$充分接近$1$时，$f(x-1)\neq 0$，上式可求导。

        由链式法则
        \[
            \begin{aligned}
                y'(x)
                &=\frac{1}{2}\,[f(x-1)^2]^{-1/2}\cdot 2f(x-1)\cdot f'(x-1)\\
                &=\frac{f(x-1)f'(x-1)}{\sqrt{[f(x-1)]^2}}.
            \end{aligned}
        \]
        代入$x=1$：
        \[
            y'(1)=\frac{f(0)f'(0)}{|f(0)|}
            =\frac{(-1)\cdot 1}{1}
            =-1.
        \]
        故答案为
        \[
            \boxed{y'(1)=-1}.
        \]
    \end{analysisbox}

    \subsubsection*{第2题解答}
    \begin{analysisbox}
        先化简：
        \[
            y=\ln\sqrt{\frac{1-x}{1+x^2}}
            =\frac12\Bigl[\ln(1-x)-\ln(1+x^2)\Bigr].
        \]
        对$x$求导：
        \[
            y'
            =\frac12\left(\frac{(1-x)'}{1-x}-\frac{(1+x^2)'}{1+x^2}\right)
            =\frac12\left(-\frac1{1-x}-\frac{2x}{1+x^2}\right).
        \]
        代入$x=0$：
        \[
            y'(0)=\frac12\left(-1-0\right)=-\frac12.
        \]
        故答案为
        \[
            \boxed{-\frac12}.
        \]
    \end{analysisbox}

    \subsubsection*{第3题解答}
    \begin{analysisbox}
        参数方程为
        \[
            x=2t+|t|,\quad y=|t|\tan t,
        \]
        分$t$的符号讨论。

        \textbf{(1) 写成显函数}

        当$t<0$时，$|t|=-t$，故
        \[
            x=t,\quad y=-t\tan t.
        \]
        由$x=t$得
        \[
            f(x)=-x\tan x,\quad x<0.
        \]

        当$t\ge 0$时，$|t|=t$，故
        \[
            x=3t,\quad y=t\tan t.
        \]
        由$t=\dfrac x3$得
        \[
            f(x)=\frac x3\tan\frac x3,\quad x\ge 0.
        \]
        因而
        \[
            f(x)=
            \begin{cases}
                -x\tan x, & x<0,\\[0.3em]
                \dfrac x3\tan\dfrac x3, & x\ge 0.
            \end{cases}
        \]

        \textbf{(2) 连续性与一阶导数}

        \[
            \lim_{x\to 0^-}(-x\tan x)=0,\qquad
            \lim_{x\to 0^+}\frac x3\tan\frac x3=0,
        \]
        且$f(0)=0$，故$f$在$0$处连续。

        对分段函数求导：
        \[
            x<0:\ f'(x)=-(\tan x+x\sec^2x),
        \]
        \[
            x>0:\ f'(x)=\frac13\tan\frac x3+\frac x9\sec^2\frac x3.
        \]
        于是
        \[
            f'_-(0)=\lim_{x\to 0^-}f'(x)=0,\qquad
            f'_+(0)=\lim_{x\to 0^+}f'(x)=0.
        \]
        故$f'(0)$存在且$f'(0)=0$，并且$f'(x)$在$0$处连续。

        \textbf{(3) 二阶导数}

        再对上式求导：
        \[
            x<0:\ f''(x)=-(2\sec^2x+2x\sec^2x\tan x),
        \]
        \[
            x>0:\ f''(x)=\frac{2}{9}\sec^2\frac x3+\frac{2x}{27}\sec^2\frac x3\tan\frac x3.
        \]
        故
        \[
            f''_-(0)=-2,\qquad f''_+(0)=\frac29,
        \]
        左右不等，故$f''(0)$不存在。

        因此正确选项为
        \[
            \boxed{\text{C}}.
        \]
    \end{analysisbox}

    \subsubsection*{第4题解答}
    \begin{analysisbox}
        在$|x|<1$内，
        \[
            \ln(1-x)=-\sum_{k=1}^{\infty}\frac{x^k}{k}.
        \]
        故
        \[
            f(x)=x^2\ln(1-x)
            =-\sum_{k=1}^{\infty}\frac{x^{k+2}}{k}
            =-\sum_{n=3}^{\infty}\frac{x^n}{n-2}\quad(n=k+2).
        \]
        由幂级数系数与导数关系：若
        \[
            f(x)=\sum_{n=0}^{\infty}a_nx^n,
        \]
        则
        \[
            f^{(n)}(0)=n!a_n.
        \]
        这里$n\ge 3$时
        \[
            a_n=-\frac1{n-2},
        \]
        所以
        \[
            f^{(n)}(0)=n!\left(-\frac1{n-2}\right)=-\frac{n!}{n-2}.
        \]
        故答案为
        \[
            \boxed{f^{(n)}(0)=-\frac{n!}{n-2}\ (n\ge 3)}.
        \]
    \end{analysisbox}

    \subsubsection*{第5题解答}
    \begin{analysisbox}
        对于固定$t$，
        \[
            f(t)=\lim_{n\to\infty}\cos t\left(\frac{n+t}{n-t}\right)^n
            =\cos t\cdot\exp\!\left(\lim_{n\to\infty}n\ln\frac{1+t/n}{1-t/n}\right).
        \]
        令
        \[
            L_n=n\left[\ln\!\left(1+\frac tn\right)-\ln\!\left(1-\frac tn\right)\right].
        \]
        由$\ln(1+u)=u+o(u)$，$\ln(1-u)=-u+o(u)$（$u\to0$）得
        \[
            L_n
            =n\left[\frac tn+o\!\left(\frac1n\right)-\left(-\frac tn+o\!\left(\frac1n\right)\right)\right]
            =2t+o(1)\to 2t.
        \]
        因而
        \[
            f(t)=e^{2t}\cos t.
        \]
        求导：
        \[
            f'(t)=e^{2t}(2\cos t-\sin t).
        \]
        代入$t=0$：
        \[
            f'(0)=2.
        \]
        故答案为
        \[
            \boxed{2}.
        \]
    \end{analysisbox}

    \subsubsection*{第6题解答}
    \begin{analysisbox}
        先说明：题面写成
        \[
            f(x)=\lim_{t\to\infty}\left(1+\frac1t\right)^{t\sin t}
        \]
        时，右端与$x$无关，且该极限本身不存在（因$\sin t$振荡），故“$f'(x)$”无从谈起。

        该类题的常见正确形式为
        \[
            f(x)=\lim_{t\to\infty}\left(1+\frac{x}{t}\right)^{t\sin x}.
        \]
        下面按该标准题型求解。取对数：
        \[
            \ln f(x)=\lim_{t\to\infty}t\sin x\cdot \ln\left(1+\frac xt\right)
            =\sin x\cdot \lim_{t\to\infty}t\ln\left(1+\frac xt\right).
        \]
        由极限
        \[
            \lim_{t\to\infty}t\ln\left(1+\frac xt\right)=x
        \]
        得
        \[
            \ln f(x)=x\sin x.
        \]
        故
        \[
            f(x)=e^{x\sin x},
        \]
        进而
        \[
            f'(x)=e^{x\sin x}\cdot (x\sin x)'=e^{x\sin x}(\sin x+x\cos x).
        \]
        故（按标准题型）答案为
        \[
            \boxed{f'(x)=e^{x\sin x}(\sin x+x\cos x)}.
        \]
    \end{analysisbox}

    \subsubsection*{第7题解答}
    \begin{analysisbox}
        因
        \[
            x^2-x=x(x-1),
        \]
        在$0<x<1$时$x^2-x<0$，即$x>x^2$；在$1<x<2$时$x^2-x>0$，即$x^2>x$。
        故
        \[
            f(x)=
            \begin{cases}
                x, & 0<x\le 1,\\[0.3em]
                x^2, & 1\le x<2.
            \end{cases}
        \]
        从而
        \[
            f'(x)=
            \begin{cases}
                1, & 0<x<1,\\
                2x, & 1<x<2,
            \end{cases}
        \]
        且
        \[
            f'_-(1)=1,\qquad f'_+(1)=2,
        \]
        左右导数不等，故$f'(1)$不存在。
    \end{analysisbox}

    \subsubsection*{第8题解答}
    \begin{analysisbox}
        \[
            f(x)=x^2(e^x+5)=x^2e^x+5x^2.
        \]
        因$5x^2$的5阶导为$0$，故仅需计算$(x^2e^x)^{(5)}$。

        设$g(x)=x^2e^x=e^x\cdot x^2$。用公式
        \[
            \frac{d^n}{dx^n}\bigl(e^x p(x)\bigr)=e^x\sum_{k=0}^n\binom{n}{k}p^{(k)}(x),
        \]
        取$p(x)=x^2,\ n=5$，并用
        \[
            p'(x)=2x,\quad p''(x)=2,\quad p^{(k)}(x)=0\ (k\ge 3),
        \]
        得
        \[
            g^{(5)}(x)=e^x\left[\binom50x^2+\binom51(2x)+\binom52(2)\right]
            =e^x(x^2+10x+20).
        \]
        故
        \[
            f^{(5)}(1)=g^{(5)}(1)=e(1+10+20)=31e.
        \]
        所以答案为
        \[
            \boxed{31e}.
        \]
    \end{analysisbox}

    \subsubsection*{第9题解答}
    \begin{analysisbox}
        \[
            f(x)=(x-1)^{10}\sin x.
        \]
        用Leibniz公式：
        \[
            f^{(11)}(x)=\sum_{k=0}^{11}\binom{11}{k}\bigl[(x-1)^{10}\bigr]^{(k)}(\sin x)^{(11-k)}.
        \]
        注意
        \[
            \bigl[(x-1)^{10}\bigr]^{(k)}\Big|_{x=1}=0\quad (k<10),\qquad
            \bigl[(x-1)^{10}\bigr]^{(11)}\equiv 0.
        \]
        因而仅$k=10$项保留：
        \[
            f^{(11)}(1)=\binom{11}{10}\cdot 10!\cdot (\sin x)'_{x=1}
            =11\cdot 10!\cos 1.
        \]
        所以答案为
        \[
            \boxed{11\cdot 10!\cos 1}.
        \]
    \end{analysisbox}

    \subsection{一元函数微分学的应用 - 几何应用}

    \begin{enumerate}
        \item 曲线$\tan(x + y + \dfrac{\pi}{4}) = e^y$在点$(0, 0)$处的切线方程为
        \item 曲线$y = (2x - 1)e^\dfrac{1}{x}$的斜渐近线方程为
        \item 设函数$y = y(x)$由方程
        \[
            2y^3 - 2y^2 + 2xy - x^2 = 1
        \]
        所确定，试求$y = y(x)$的驻点，并判别它是否为极值点
    \end{enumerate}

    \subsubsection{解答}

    \subsubsection*{第1题解答}
    \begin{analysisbox}
        由
        \[
            \tan\!\left(x+y+\frac{\pi}{4}\right)=e^y
        \]
        两边对$x$求导，得
        \[
            \sec^2\!\left(x+y+\frac{\pi}{4}\right)(1+y')=e^y\,y'.
        \]
        在点$(0,0)$处，
        \[
            \sec^2\!\left(\frac{\pi}{4}\right)=2,\quad e^0=1,
        \]
        故
        \[
            2(1+y')=y' \Longrightarrow y'=-2.
        \]
        因而切线方程为
        \[
            y-0=-2(x-0),
        \]
        即
        \[
            \boxed{y=-2x}.
        \]
    \end{analysisbox}

    \subsubsection*{第2题解答}
    \begin{analysisbox}
        设斜渐近线为$y=kx+b$，其中
        \[
            k=\lim_{x\to\infty}\frac{y}{x}
            =\lim_{x\to\infty}\frac{(2x-1)e^{\dfrac{1}{x}}}{x}
            =\lim_{x\to\infty}\left(2-\frac{1}{x}\right)e^{\dfrac{1}{x}}=2.
        \]
        再求
        \[
            b=\lim_{x\to\infty}(y-kx)
            =\lim_{x\to\infty}\left[(2x-1)e^{\dfrac{1}{x}}-2x\right].
        \]
        利用展开式$e^{\dfrac{1}{x}}=1+\dfrac{1}{x}+o\!\left(\dfrac{1}{x}\right)$，有
        \[
            (2x-1)e^{\dfrac{1}{x}}
            =(2x-1)\left(1+\frac{1}{x}+o\!\left(\frac{1}{x}\right)\right)
            =2x+1+o(1),
        \]
        从而$b=1$。
        因此斜渐近线为
        \[
            \boxed{y=2x+1}.
        \]
    \end{analysisbox}

    \subsubsection*{第3题解答}
    \begin{analysisbox}
        由
        \[
            2y^3-2y^2+2xy-x^2=1
        \]
        对$x$求导，得
        \[
            6y^2y'-4yy'+2y+2xy'-2x=0.
        \]
        驻点满足$y'=0$，代入上式立得
        \[
            2y-2x=0 \Longrightarrow x=y.
        \]
        再代回原方程
        \[
            2y^3-2y^2+2y^2-y^2=1
            \Longrightarrow 2y^3-y^2-1=0
            \Longrightarrow (y-1)(2y^2+y+1)=0.
        \]
        因$2y^2+y+1>0$，得唯一实根$y=1$，故$x=1$。
        所以驻点为
        \[
            \boxed{(1,1)}.
        \]

        再对
        \[
            6y^2y'-4yy'+2y+2xy'-2x=0
        \]
        两边对$x$求导，得
        \[
            (12y-4)(y')^2+(6y^2-4y+2x)y''+4y'-2=0.
        \]
        将$x=1,\ y=1,\ y'=0$代入：
        \[
            4y''-2=0 \Longrightarrow y''=\frac12>0.
        \]
        因此$(1,1)$为极小值点。

        \medskip
        结论：函数$y=y(x)$仅有一个驻点$\boxed{(1,1)}$，且该点是$\boxed{\text{极小值点}}$。
    \end{analysisbox}

    \subsection{一元函数微分学的应用 - 中值定理、微分等式与微分不等式}

    \begin{enumerate}
        \item 若函数$f(x) = \dfrac{1}{xe^{-1} - a}$在$(-\infty, +\infty)$内处处连续，则常数$a$的取值范围为
        \item 设实数$a_0, a_1, a_2, \cdots, a_n$满足$a_0 + \dfrac{a_1}{2} + \cdots + \dfrac{a_n}{n+1} = 0$，证明：方程$a_0 + a_{1}x + a_{2}x^2 + \cdots + a_{n}a^n = 0$在$(0, 1)$内至少有一个根
        \item 已知$f(x)$在$[0, 1]$上连续，在$(0, 1)$内可导，且$f(0) = 0, f(1) = 1$。证明：
        \begin{itemize}
            \item 存在$\varepsilon \in (0, 1)$，使得$f(\varepsilon) = 1 - \varepsilon$
            \item 存在$\eta, \tau \in (0, 1)$，使得$f'(\eta)f'(\tau) = 1$
        \end{itemize}
        \item 设函数$f(x)$在$[a, b]$上连续$(0 < a < b)$，在$(a, b)$内可导，且$f(a) \neq f(b)$。证明：存在$\varepsilon, \eta \in (a, b)$，使得$\dfrac{f'(\varepsilon)}{2\varepsilon} = \dfrac{f'(\eta)}{b + a}$
        \item 设$f(x)$在$[0, 1]$上具有二阶导数，且满足条件$|f(x)| \leq a, |f''(x)| \leq b$，其中$a, b$都是非负常数，$c$是$(0, 1)$内任意一点
        \begin{itemize}
            \item 写出$f(x)$在点$x = c$处带拉格朗日型余项的一阶泰勒公式
            \item 证明$|f'(x)| \leq 2a + \dfrac{b}{2}$
        \end{itemize}
        \item 证明$0 < x < \dfrac{\pi}{4}$时，有$\tan x < \dfrac{4}{\pi}x$
        \item 设$p, q$是大于$1$的常数，并且$\dfrac{1}{p} + \dfrac{1}{q} = 1$。证明：对于任意的$x > 0$，有$\dfrac{1}{p}x^p + \dfrac{1}{q} \geq x$
    \end{enumerate}

    \subsubsection{解答}

    \subsubsection*{第1题解答}
    \begin{analysisbox}
        \[
            f(x)=\frac{1}{xe^{-x}-a}.
        \]
        要使$f(x)$在$\mathbb R$上处处连续，必须且只需
        \[
            xe^{-x}-a\neq 0\quad(\forall x\in\mathbb R),
        \]
        即$a$不属于函数$\varphi(x)=xe^{-x}$的值域。

        求$\varphi$值域：
        \[
            \varphi'(x)=e^{-x}(1-x).
        \]
        故$\varphi$在$(-\infty,1)$单调增、在$(1,+\infty)$单调减，且
        \[
            \varphi(1)=\frac1e,\quad
            \lim_{x\to-\infty}xe^{-x}=-\infty,\quad
            \lim_{x\to+\infty}xe^{-x}=0.
        \]
        因而
        \[
            \mathrm{Range}(\varphi)=(-\infty,\tfrac1e].
        \]
        为使分母恒不为零，须有
        \[
            a\notin(-\infty,\tfrac1e]\Longleftrightarrow a>\frac1e.
        \]

        \[
            \boxed{a>\frac1e}.
        \]
    \end{analysisbox}

    \subsubsection*{第2题解答}
    \begin{analysisbox}
        设辅助函数
        \[
            F(x)=a_0x+\frac{a_1}{2}x^2+\cdots+\frac{a_n}{n+1}x^{n+1},\quad x\in[0,1].
        \]
        则$F$在$[0,1]$连续、在$(0,1)$可导，且由题设
        \[
            F(1)-F(0)=a_0+\frac{a_1}{2}+\cdots+\frac{a_n}{n+1}=0.
        \]
        故$F(1)=F(0)$，由罗尔定理，存在$\xi\in(0,1)$使
        \[
            F'(\xi)=0.
        \]
        又
        \[
            F'(x)=a_0+a_1x+\cdots+a_nx^n.
        \]
        所以
        \[
            a_0+a_1\xi+\cdots+a_n\xi^n=0,
        \]
        即方程
        \[
            a_0+a_1x+\cdots+a_nx^n=0
        \]
        在$(0,1)$内至少有一根。
    \end{analysisbox}

    \subsubsection*{第3题解答}
    \begin{analysisbox}
        \textbf{(1) 证存在$\varepsilon\in(0,1)$使$f(\varepsilon)=1-\varepsilon$}

        令
        \[
            \phi(x)=f(x)+x-1.
        \]
        则$\phi$在$[0,1]$上连续，且
        \[
            \phi(0)=f(0)-1=-1<0,\qquad
            \phi(1)=f(1)+1-1=1>0.
        \]
        由介值定理，存在$\varepsilon\in(0,1)$使$\phi(\varepsilon)=0$，即
        \[
            f(\varepsilon)=1-\varepsilon.
        \]

        \textbf{(2) 证存在$\eta,\tau\in(0,1)$使$f'(\eta)f'(\tau)=1$}

        由上一步$\varepsilon\in(0,1)$，在区间$[0,\varepsilon]$上应用拉格朗日中值定理，存在$\eta\in(0,\varepsilon)$使
        \[
            f'(\eta)=\frac{f(\varepsilon)-f(0)}{\varepsilon-0}
            =\frac{1-\varepsilon}{\varepsilon}.
        \]
        在区间$[\varepsilon,1]$上应用拉格朗日中值定理，存在$\tau\in(\varepsilon,1)$使
        \[
            f'(\tau)=\frac{f(1)-f(\varepsilon)}{1-\varepsilon}
            =\frac{1-(1-\varepsilon)}{1-\varepsilon}
            =\frac{\varepsilon}{1-\varepsilon}.
        \]
        因而
        \[
            f'(\eta)f'(\tau)
            =\frac{1-\varepsilon}{\varepsilon}\cdot\frac{\varepsilon}{1-\varepsilon}
            =1.
        \]
    \end{analysisbox}

    \subsubsection*{第4题解答}
    \begin{analysisbox}
        由拉格朗日中值定理，存在$\eta\in(a,b)$使
        \[
            f'(\eta)=\frac{f(b)-f(a)}{b-a}.
        \]
        再在$[a,b]$上对函数$f(x)$与$g(x)=x^2$应用柯西中值定理，存在$\varepsilon\in(a,b)$使
        \[
            \frac{f'(\varepsilon)}{g'(\varepsilon)}
            =\frac{f(b)-f(a)}{g(b)-g(a)}
            =\frac{f(b)-f(a)}{b^2-a^2}.
        \]
        即
        \[
            \frac{f'(\varepsilon)}{2\varepsilon}
            =\frac{f(b)-f(a)}{(b-a)(a+b)}
            =\frac{f'(\eta)}{a+b}.
        \]
        因此存在$\varepsilon,\eta\in(a,b)$使
        \[
            \boxed{\frac{f'(\varepsilon)}{2\varepsilon}=\frac{f'(\eta)}{a+b}}.
        \]
    \end{analysisbox}

    \subsubsection*{第5题解答}
    \begin{analysisbox}
        \textbf{(1)} 对任意$x\in[0,1]$，存在$\xi_x$介于$x$与$c$之间，使
        \[
            f(x)=f(c)+f'(c)(x-c)+\frac{f''(\xi_x)}{2}(x-c)^2.
        \]

        \textbf{(2)} 分别令$x=0,1$，得存在$\xi_1\in(0,c),\xi_2\in(c,1)$使
        \[
            f(0)=f(c)-cf'(c)+\frac{f''(\xi_1)}{2}c^2,
        \]
        \[
            f(1)=f(c)+(1-c)f'(c)+\frac{f''(\xi_2)}{2}(1-c)^2.
        \]
        两式相减：
        \[
            f'(c)=f(1)-f(0)-\frac{f''(\xi_2)}{2}(1-c)^2+\frac{f''(\xi_1)}{2}c^2.
        \]
        取绝对值并利用$\lvert f(x)\rvert\le a,\lvert f''(x)\rvert\le b$：
        \[
            \begin{aligned}
                |f'(c)|
                &\le |f(1)|+|f(0)|+\frac{|f''(\xi_2)|}{2}(1-c)^2+\frac{|f''(\xi_1)|}{2}c^2\\
                &\le 2a+\frac b2\bigl[(1-c)^2+c^2\bigr]
            \end{aligned}
        \]
        又因$c \in (0, 1)$，有$(1-c)^2+c^2 \leq 1$，故
        \[
            |f'(c)| \leq 2a + \frac{b}{2}.
        \]
        又
        \[
            f(0) = f(1) + f'(1)(0-1) + \dfrac{f''(\varepsilon_3)}{2!}(0-1)^2, 0 < \varepsilon_3 < 1
        \]
        \[
            f(1) = f(0) + f'(0)(1-0) + \dfrac{f''(\varepsilon_4)}{2!}(1-0)^2, 0 < \varepsilon_4 < 1
        \]
        故
        \[
            f'(1) = f(1)-f(0)+\frac{f''(\varepsilon_3)}{2},
            \quad
            f'(0) = f(1)-f(0)-\frac{f''(\varepsilon_4)}{2},
        \]
        \[
            |f'(0)| \le |f(1)|+|f(0)|+\frac{|f''(\varepsilon_4)|}{2}\le 2a+\frac b2,
        \]
        \[
            |f'(1)| \le |f(1)|+|f(0)|+\frac{|f''(\varepsilon_3)|}{2}\le 2a+\frac b2.
        \]

        综上所述
        \[
            \boxed{|f'(x)|\le 2a+\frac b2,\ \forall x\in[0,1].}
        \]
    \end{analysisbox}

    \subsubsection*{第6题解答}
    \begin{analysisbox}
        设辅助函数
        \[
            \Phi(x)=\tan x-\frac4\pi x,\qquad x\in\left[0,\frac\pi4\right].
        \]
        有
        \[
            \Phi(0)=0,\qquad \Phi\!\left(\frac\pi4\right)=1-1=0.
        \]
        再求二阶导数：
        \[
            \Phi''(x)=2\sec^2x\tan x>0,\quad 0<x<\frac\pi4.
        \]
        故$\Phi$在$\left[0,\dfrac\pi4\right]$上严格凸。由凸函数性质，对任意$0<t<1$，
        \[
            \Phi\!\left(t\cdot 0+(1-t)\frac\pi4\right)
            <t\Phi(0)+(1-t)\Phi\!\left(\frac\pi4\right)=0.
        \]
        即对$0<x<\dfrac\pi4$有$\Phi(x)<0$，从而
        \[
            \tan x-\frac4\pi x<0
            \Longrightarrow
            \boxed{\tan x<\frac4\pi x,\quad 0<x<\frac\pi4}.
        \]
    \end{analysisbox}

    \subsubsection*{第7题解答}
    \begin{analysisbox}
        设
        \[
            \Phi(x)=\frac1p x^p+\frac1q-x,\qquad x>0.
        \]
        由$\dfrac1p+\dfrac1q=1$，且$p>1$，可得
        \[
            \Phi'(x)=x^{p-1}-1,\qquad
            \Phi''(x)=(p-1)x^{p-2}>0.
        \]
        故$\Phi$为严格凸函数，驻点由$\Phi'(x)=0$得$x=1$，且该点为全局最小点。

        计算最小值：
        \[
            \Phi(1)=\frac1p+\frac1q-1=0.
        \]
        因此对任意$x>0$有
        \[
            \Phi(x)\ge 0,
        \]
        即
        \[
            \boxed{\frac1p x^p+\frac1q\ge x}.
        \]
    \end{analysisbox}

    \begin{enumerate}
        \item 质点$P$沿抛物线$x = y^2(y > 0)$移动，$P$的横坐标$x$对时间的变化率为$5\text{ cm/s}$，当$x = 9$时，点$P$到原点$O$的距离对时间的变化率为
        \item 甲车以$24km/h$的速度向北行驶，同时正东$10km$处乙车以$20km/h$的速度向东行驶，从这一时刻起经过$1$小时后，求两车间的距离对时间的变化率
    \end{enumerate}

    \subsubsection{解答}

    \subsubsection*{第1题解答}
    \begin{analysisbox}
        设点$P(x,y)$到原点距离为
        \[
            r=\sqrt{x^2+y^2}.
        \]
        两边对$t$求导得
        \[
            \frac{dr}{dt}=\frac{d}{dt}\sqrt{x^2 + y^2} = \frac{d}{dx}\sqrt{x^2 + x} \cdot \frac{dx}{dt} = \frac{5(2x + 1)}{2\sqrt{x^2 + x}}
        \]
        当$x=9$时
        \[
            \frac{dr}{dt}\Big|_{x=9}
            =\frac{5(2x + 1)}{2\sqrt{x^2 + x}}\Big|_{x=9} = \boxed{\frac{95}{6\sqrt{10}}\ \text{cm/s}}
        \]
    \end{analysisbox}

    \subsubsection*{第2题解答}
    \begin{analysisbox}
        设两车在$t$小时后的南北向距离为$y$，东西向距离为$x$，两车间距离为$s(t)$，则
        \[
            s(t)=\sqrt{x^2+y^2}.
        \]
        由题意可得
        \[
            x=10+20t,\quad y=24t,\quad \frac{dx}{dt}=20,\quad \frac{dy}{dt}=24.
        \]
        两边对$t$求导：
        \[
            \frac{ds}{dt} = \frac{d}{dt}\sqrt{x^2 + y^2} = \frac{x\dfrac{dx}{dt} + y\dfrac{dy}{dt}}{\sqrt{x^2 + y^2}}
        \]
        当$t=1$时，$x = 30,\ y = 24$
        \[
            \frac{ds}{dt}\Big|_{t=1}
            =\frac{30\cdot 20+24\cdot 24}{\sqrt{30^2+24^2}}
            =\boxed{\frac{196}{\sqrt{41}}\ \text{km/h}}.
        \]
    \end{analysisbox}
