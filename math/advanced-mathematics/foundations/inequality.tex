    \section{不等式}

    \subsection{公式}

    \begin{enumerate}
        \item 若$b > a > 0$，则$a \leq |x| < b \Leftrightarrow a \leq x < b \,\text{或}-b < x \leq -a$
        \item 均值不等式
        \[
            \begin{aligned}
                \frac{2}{\displaystyle\frac{1}{a}+\displaystyle\frac{1}{b}}
                &\le \sqrt{ab}
                \le \frac{a+b}{2}
                \le \sqrt{\frac{a^2+b^2}{2}},
                \qquad a,b>0
            \end{aligned}
        \]

        \[
            \text{调和平均值}
            \;\leq\;
            \text{几何平均值}
            \;\leq\;
            \text{算术平均值}
            \;\leq\;
            \text{均方根}
        \]
        \item $a^2 + b^2 \geq 2ab$
        \begin{itemize}
            \item $\displaystyle \frac{a + b}{2} \geq \sqrt{ab}, \qquad a,b > 0$
            \item $\displaystyle \frac{a}{b} \leq \frac{a^2 + \displaystyle \frac{1}{b^2}}{2}$
            \item $\displaystyle a + \frac{1}{a} \geq 2, \qquad a>0$
            \item $\displaystyle \frac{a + b + c}{3} \geq \sqrt[3]{abc}$
        \end{itemize}
        \item 三角不等式
        \begin{itemize}
            \item 基本不等式：
            \[
                \bigl||a| - |b|\bigr| \le |a \pm b| \le |a| + |b|
            \]
            \item 等号条件：
            \begin{align*}
                \bigl||a| - |b|\bigr| &= |a + b| &\Leftrightarrow\quad & ab \le 0,\\
                |a + b| &= |a| + |b| &\Leftrightarrow\quad & ab \ge 0,\\
                \bigl||a| - |b|\bigr| &= |a - b| &\Leftrightarrow\quad & ab \ge 0,\\
                |a - b| &= |a| + |b| &\Leftrightarrow\quad & ab \le 0.
            \end{align*}
        \end{itemize}

        \item 柯西不等式:
        \[
            \left(a_1^2 + a_2^2 + \cdots + a_n^2\right)
            \left(b_1^2 + b_2^2 + \cdots + b_n^2\right)
            \geq
            \left(a_1 b_1 + a_2 b_2 + \cdots + a_n b_n\right)^2
        \]

        令 \( b_1 = b_2 = \cdots = b_n = 1 \)，则有
        \[
            a_1^2 + a_2^2 + \cdots + a_n^2
            \geq
            \frac{(a_1 + a_2 + \cdots + a_n)^2}{n}
        \]
        注: $\color{red}{(a_1^2 + a_2^2)(b_1^2 + b_2^2) \geq (a_{1}b_{1} + a_2 b_2)^2}$
        \item Bernoulli（伯努利）不等式: $(1+x)^n \ge 1 + nx, x > -1$
        \item 若 $0 < a < x < b, 0 < c < y < d$，则$\displaystyle\frac{c}{b} < \frac{y}{x} < \frac{d}{a}$
        \item $\displaystyle\frac{2}{\pi}x < \sin x < x < \tan x,\quad x \in \left(0,\displaystyle\frac{\pi}{2}\right)$
        \item $\sin x < x, \quad x > 0$
        \item $\color{red}{\bigstar}$当 $0 < x < \displaystyle\frac{\pi}{4}$时，$x < \tan x < \displaystyle\frac{4}{\pi}x$
        \item $\arctan x \leq x \leq \arcsin x, \quad 0 \leq x \leq 1$
        \item $e^x \geq x + 1, \quad x\text{为任意值}$，$y = x + 1$是$y = e^x$在$x = 0$处的切线方程
        \item $x > 0$时，$\color{red}{x - 1 \geq \ln x}\;$或$\;x \geq \ln x + 1$
        \item
        \[
            \frac{1}{1+x}
            <
            \ln\!\left(1+\frac{1}{x}\right)
            <
            \frac{1}{x},
            \quad x>0,
        \]
        或等价地，
        \[
            \frac{x}{1+x}
            \le
            \ln(1+x)
            <
            x,
            \quad x>0.
        \]
        \begin{analysisbox}[证明]
            令$f(t) = \ln t$，在$[x, 1+x]$用拉格朗日中值定理
        \end{analysisbox}
    \end{enumerate}
