    \section{二重积分}

    \subsection{概念、性质与对称性}

    \subsubsection{概念}

    设 $f(x,y)$ 是有界闭区域 $D$ 上的有界函数，将 $D$ 任意分成 $n$ 个小闭区域 $\Delta\sigma_1, \Delta\sigma_2, \cdots, \Delta\sigma_n$，其中 $\Delta\sigma_i$ 既表示第 $i$ 个小闭区域，也表示它的面积。在每个 $\Delta\sigma_i$ 上任取一点 $(\xi_i, \eta_i)$，作和
    \[
        \sum_{i=1}^{n} f(\xi_i, \eta_i) \Delta\sigma_i
    \]
    如果当各小闭区域的直径中的最大值 $\lambda \to 0$ 时，这和的极限总存在，则称此极限为函数 $f(x,y)$ 在闭区域 $D$ 上的\textbf{二重积分}，记为
    \[
        \iint_D f(x,y) \mathrm{d}\sigma = \lim_{\lambda \to 0} \sum_{i=1}^{n} f(\xi_i, \eta_i) \Delta\sigma_i
    \]
    其中 $f(x,y)$ 叫做\textbf{被积函数}，$f(x,y)\mathrm{d}\sigma$ 叫做\textbf{被积表达式}，$\mathrm{d}\sigma$ 叫做\textbf{面积元素}，$x$ 与 $y$ 叫做\textbf{积分变量}，$D$ 叫做\textbf{积分区域}。

    \begin{analysisbox}[几何意义]
        如果 $f(x,y) \geq 0$，则二重积分 $\displaystyle\iint_D f(x,y) \mathrm{d}\sigma$ 表示以 $D$ 为底、以曲面 $z = f(x,y)$ 为顶的曲顶柱体的体积。
    \end{analysisbox}

    \subsubsection{性质}

    \begin{enumerate}
        \item \textbf{求区域面积}：$\displaystyle\iint_D 1 \mathrm{d}\sigma = \iint_D \mathrm{d}\sigma = S_D$（区域 $D$ 的面积）

        \item \textbf{可积函数必有界}：若 $f(x,y)$ 在 $D$ 上可积，则 $f(x,y)$ 在 $D$ 上有界

        \item \textbf{积分的线性性质}：设 $\alpha, \beta$ 为常数，则
        \[
            \iint_D [\alpha f(x,y) + \beta g(x,y)] \mathrm{d}\sigma = \alpha \iint_D f(x,y) \mathrm{d}\sigma + \beta \iint_D g(x,y) \mathrm{d}\sigma
        \]

        \item \textbf{积分的可加性}：如果闭区域 $D$ 被有限条曲线分为有限个部分闭区域，则在 $D$ 上的二重积分等于在各部分闭区域上的二重积分的和。设 $D = D_1 \cup D_2$（$D_1$ 与 $D_2$ 无公共内点），则
        \[
            \iint_D f(x,y) \mathrm{d}\sigma = \iint_{D_1} f(x,y) \mathrm{d}\sigma + \iint_{D_2} f(x,y) \mathrm{d}\sigma
        \]

        \item \textbf{积分的保号性}：
        \begin{circlenum}
            \item 若在 $D$ 上 $f(x,y) \geq 0$，则 $\displaystyle\iint_D f(x,y) \mathrm{d}\sigma \geq 0$
            \item 若在 $D$ 上 $f(x,y) \leq g(x,y)$，则 $\displaystyle\iint_D f(x,y) \mathrm{d}\sigma \leq \iint_D g(x,y) \mathrm{d}\sigma$
            \item $\displaystyle\left| \iint_D f(x,y) \mathrm{d}\sigma \right| \leq \iint_D |f(x,y)| \mathrm{d}\sigma$
        \end{circlenum}

        \item \textbf{二重积分的估值定理}：设 $M$ 和 $m$ 分别是 $f(x,y)$ 在闭区域 $D$ 上的最大值和最小值，$S_D$ 是 $D$ 的面积，则
        \[
            mS_D \leq \iint_D f(x,y) \mathrm{d}\sigma \leq MS_D
        \]

        \item \textbf{二重积分的中值定理}：设 $f(x,y)$ 在闭区域 $D$ 上连续，$S_D$ 是 $D$ 的面积，则在 $D$ 上至少存在一点 $(\xi, \eta)$，使得
        \[
            \iint_D f(x,y) \mathrm{d}\sigma = f(\xi, \eta) S_D
        \]
    \end{enumerate}

    \subsubsection{对称性}

    \begin{enumerate}
        \item \textbf{普通对称性}：设区域 $D$ 关于 $x = a$ 对称（即 $(2a-x, y) \in D$），则
        \[
            \iint_D f(x,y) \mathrm{d}\sigma = \begin{cases}
                2\iint_{D_1} f(x,y) \mathrm{d}\sigma, & f(2a-x, y) = f(x,y) \\
                0, & f(2a-x, y) = -f(x,y)
            \end{cases}
        \]
        其中 $D_1$ 为 $D$ 在 $x \geq a$（或 $x \leq a$）一侧的部分。

        设区域 $D$ 关于 $y = a$ 对称（即 $(x, 2a-y) \in D$），则
        \[
            \iint_D f(x,y) \mathrm{d}\sigma = \begin{cases}
                2\iint_{D_1} f(x,y) \mathrm{d}\sigma, & f(x, 2a-y) = f(x,y) \\
                0, & f(x, 2a-y) = -f(x,y)
            \end{cases}
        \]

        \item \textbf{关于原点对称}：设区域 $D$ 关于原点对称（即 $(-x, -y) \in D$），则
        \[
            \iint_D f(x,y) \mathrm{d}\sigma = \begin{cases}
                2\iint_{D_1} f(x,y) \mathrm{d}\sigma, & f(-x,-y) = f(x,y) \\
                0, & f(-x,-y) = -f(x,y)
            \end{cases}
        \]

        \item \textbf{关于 $y = -x$ 对称}：设区域 $D$ 关于 $y = -x$ 对称（即 $(-y, -x) \in D$），则
        \[
            \iint_D f(x,y) \mathrm{d}\sigma = \iint_D f(-y,-x) \mathrm{d}\sigma
        \]

        \item \textbf{轮换对称性}：若 $D$ 关于 $y = x$ 对称（即 $(y, x) \in D$），则
        \[
            \iint_D f(x,y) \mathrm{d}\sigma = \iint_D f(y,x) \mathrm{d}\sigma
        \]
        特别地，$\displaystyle\iint_D f(x) \mathrm{d}\sigma = \iint_D f(y) \mathrm{d}\sigma$
    \end{enumerate}

    \subsection{计算}

    \subsubsection{直角坐标系下的计算方法}

    \begin{enumerate}
        \item \textbf{X 型区域}（先 $y$ 后 $x$）：若积分区域 $D$ 可表示为
        \[
            D = \{(x,y) \mid a \leq x \leq b, \varphi_1(x) \leq y \leq \varphi_2(x)\}
        \]
        则
        \[
            \iint_D f(x,y) \mathrm{d}x\mathrm{d}y = \int_a^b \mathrm{d}x \int_{\varphi_1(x)}^{\varphi_2(x)} f(x,y) \mathrm{d}y
        \]

        \item \textbf{Y 型区域}（先 $x$ 后 $y$）：若积分区域 $D$ 可表示为
        \[
            D = \{(x,y) \mid c \leq y \leq d, \psi_1(y) \leq x \leq \psi_2(y)\}
        \]
        则
        \[
            \iint_D f(x,y) \mathrm{d}x\mathrm{d}y = \int_c^d \mathrm{d}y \int_{\psi_1(y)}^{\psi_2(y)} f(x,y) \mathrm{d}x
        \]
    \end{enumerate}

    \begin{analysisbox}[积分次序的选择]
        \begin{circlenum}
            \item 观察区域 $D$ 的边界曲线特点
            \item 考虑被积函数的特点（哪个变量先积分更容易）
            \item 有时需要交换积分次序才能计算
        \end{circlenum}
    \end{analysisbox}

    \subsubsection{极坐标系下的计算方法}

    在极坐标系下，$\mathrm{d}\sigma = r\mathrm{d}r\mathrm{d}\theta$。

    \begin{enumerate}
        \item \textbf{极点在区域外}：若 $D = \{(r,\theta) \mid \alpha \leq \theta \leq \beta, r_1(\theta) \leq r \leq r_2(\theta)\}$，则
        \[
            \iint_D f(r\cos\theta, r\sin\theta) r\mathrm{d}r\mathrm{d}\theta = \int_\alpha^\beta \mathrm{d}\theta \int_{r_1(\theta)}^{r_2(\theta)} f(r\cos\theta, r\sin\theta) r\mathrm{d}r
        \]

        \item \textbf{极点在区域边界上}：若 $D = \{(r,\theta) \mid \alpha \leq \theta \leq \beta, 0 \leq r \leq r(\theta)\}$，则
        \[
            \iint_D f(r\cos\theta, r\sin\theta) r\mathrm{d}r\mathrm{d}\theta = \int_\alpha^\beta \mathrm{d}\theta \int_0^{r(\theta)} f(r\cos\theta, r\sin\theta) r\mathrm{d}r
        \]

        \item \textbf{极点在区域内}：若 $D = \{(r,\theta) \mid 0 \leq \theta \leq 2\pi, 0 \leq r \leq r(\theta)\}$，则
        \[
            \iint_D f(r\cos\theta, r\sin\theta) r\mathrm{d}r\mathrm{d}\theta = \int_0^{2\pi} \mathrm{d}\theta \int_0^{r(\theta)} f(r\cos\theta, r\sin\theta) r\mathrm{d}r
        \]
    \end{enumerate}

    \subsubsection{极坐标系与直角坐标系的互相转化}

    \begin{enumerate}
        \item \textbf{直角坐标 $\to$ 极坐标}：
        \[
            x = r\cos\theta, \quad y = r\sin\theta, \quad \mathrm{d}x\mathrm{d}y = r\mathrm{d}r\mathrm{d}\theta
        \]

        \item \textbf{极坐标 $\to$ 直角坐标}：
        \[
            r = \sqrt{x^2 + y^2}, \quad \theta = \arctan\frac{y}{x}
        \]
    \end{enumerate}

    \begin{analysisbox}[何时使用极坐标]
        \begin{circlenum}
            \item 积分区域 $D$ 是圆域、圆环、扇形或其部分
            \item 被积函数含有 $x^2 + y^2$、$\frac{y}{x}$ 等形式
            \item 边界曲线用极坐标表示更简单
        \end{circlenum}
    \end{analysisbox}

    \subsubsection{换元法}

    设 $f(x,y)$ 在 $D$ 上连续，变换 $\begin{cases} x = x(u,v) \\ y = y(u,v) \end{cases}$ 将 $uOv$ 平面上的闭区域 $D'$ 变为 $xOy$ 平面上的闭区域 $D$，且雅可比行列式
    \[
        J = \frac{\partial(x,y)}{\partial(u,v)} = \begin{vmatrix}
            \dfrac{\partial x}{\partial u} & \dfrac{\partial x}{\partial v} \\[6pt]
            \dfrac{\partial y}{\partial u} & \dfrac{\partial y}{\partial v}
        \end{vmatrix} \neq 0
    \]
    则有
    \[
        \iint_D f(x,y) \mathrm{d}x\mathrm{d}y = \iint_{D'} f[x(u,v), y(u,v)] |J| \mathrm{d}u\mathrm{d}v
    \]

    \begin{analysisbox}[常用换元]
        \begin{circlenum}
            \item \textbf{极坐标变换}：$x = r\cos\theta, y = r\sin\theta$，$|J| = r$
            \item \textbf{广义极坐标}：$x = ar\cos\theta, y = br\sin\theta$（适用于椭圆区域），$|J| = abr$
        \end{circlenum}
    \end{analysisbox}
